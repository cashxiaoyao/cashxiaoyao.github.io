# Grid布局的不同方式

# grid布局有一个坑，在盒子居中的时候不能像flex布局一样，

# 使用 align-items :center;，而要使用align-content:center;

# 第一种布局的css

.box {
    display:grid;
    color :#fff;
    padding :20px;
    grid-template-columns :repeat(3,143px);
    grid-template-rows :repeat(4,84px);
    /* grid-auto-flow: column;
    align-items :center; */
    /**/
    align-content:center;
    justify-content :center;
    /* justify-items:center; */
    gap:10px;
}
.box>i {
    text-align:center;
    /* // line-height :84px; */
    background-color:#fa6b5b;
}
.box>i:nth-child(10) {
    grid-column-start:1;
    grid-column-end :span 2;
}
.box>i:nth-child(12) {
    width:134px;
    height:84px;
    grid-row-start:1;
    grid-row-end:2;
    grid-column-start :4;
}
.box>i:nth-child(13) {
    grid-row-start:2;
    grid-row-end:5;
    grid-column-start :4;
}

# 第二种布局的css

.content {
    display:grid;
    color :#fff;
    padding :20px;
    grid-template-columns :repeat(4,79px);
    grid-template-rows :repeat(2,79px);
    /* // grid-auto-flow: column;
    // align-items :center; */
    align-content:center;
    justify-content :center;
    /* // justify-item:center; */
    gap:10px;
    grid-auto-rows:79px;
}
.content>i {
    background-color:blue;
}
.content>i:first-child {
    grid-row-start:1;
    grid-column-start:1;
    grid-row-end:3;
    grid-column-end:3;
}
.content>i:nth-child(4) {
    width:79px;
    height:79px;
    grid-column-start :3;
    grid-column-end:4;
    grid-row: 2/3;
}
.content>i:nth-child(5) {
    grid-column-start :4;
    grid-column-end :5;
    grid-row-start :2;
}

# 第三种布局方式的css

.bordard {
    display:grid;
    color :#000;
    padding :20px;
    grid-template-columns :repeat(3,79px);
    grid-template-rows :repeat(4,79px);
    /* // grid-auto-flow: column;
    // align-items :center; */
    align-content :center;
    justify-content :center;
    /* // justify-items center; */
    grid-template-areas :'a b c''d e f''g h i''. k .';
    gap:10px;
}
.bordard>i {
    background-color:#ccc;
}
.bordard>i:first-child{
    grid-area: k;
}
